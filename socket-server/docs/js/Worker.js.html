let Namespace = require('./Namespace');

/**
 * @desc Wrapper for current child processes' instantiated namespaces
 *
 * @author Joseph Dan Alinsug <josephdanalinsug@hotmail>
 * @version 1.0
 *
 * @param io
 * @param managerSocket
 * @returns {*}
 * @constructor
 */
let Worker = (io, managerSocket) => {

    return {
        // socket io instance
        io,
        // Manager's socket io
        managerSocket,
        // games registered
        namespaces: {},

        /**
         * @method createNamespace
         *
         * @desc Creates socket io namespace and stores it
         *
         * @method createNamespace
         * @param name
         * @param managerSocket
         * @param enableRooms
         * @memberof Worker
         * @returns {*}
         */
        createNamespace (name, enableRooms, managerSocket) {
            if (this.namespaces[name]) {
                return;
            }

            this.namespaces[name] = new Namespace(this.io.of(name), name, enableRooms, managerSocket);
            return this.namespaces[name];
        }
    };
};
module.exports = Worker;
