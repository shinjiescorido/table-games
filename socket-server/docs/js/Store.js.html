    /**
     * Flatten Object
     *
     * Flatten object paths into a string for saving into redis store
     *
     * @param data
     * @param path
     */
    dehydrate (data, path = '') {
        let flat = [];

        _.forEach(data, (value, key) => {
            let tempPath = `${path}${path ? ':' : ''}${key}`;
            if (typeof value === 'object') {
                flat = flat.concat(this.dehydrate(value, tempPath));
                return;
            }

            flat.push([tempPath, value]);
        });

        return flat;
    },

     * Rebuild dehydrated object into JavaScript equivalent object
     *
     * @param data

    },

    /**
     * Save object to redis
     *
     * @param key
     * @param object
     * @returns {Array|{index: number, input: string}|*}
     */
    save (key, object) {
        let temp = {};
        _.forEach(this.dehydrate(object), (value) => {
            temp[value[0]] = value[1];
        });

        return this.set(key, temp);
    },

    /**
     * Set data to redis store
     *
     * @param hash
     * @param key
     * @param data
     */
    set (hash, key, data) {
        return redis.multi().hmset(hash, key, data).exec();
    },


    /**
     * Set data to redis store
     *
     * @param hash
     * @param key
     * @param data
     */
    increment (hash, key, data) {
        return redis.hincrby(hash, key, data);
    },

    /**
     * Return rehydrated object
     *
     */
    get (hash, key) {
        return redis.hget(hash, key)
    }
};
